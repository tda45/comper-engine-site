Browser API Genel Bakış

Comper Engine Browser API, modern tarayıcı özellikleri için güçlü bir arayüz sağlar. Sekme yönetimi, yer işaretleri, geçmiş ve daha fazlası.

Temel Kullanım

#include "Includes/Core/comper_browser.h"

using namespace ComperCore;

int main() {
    Browser browser;
    
    // Browser'ı başlat
    if (browser.Initialize()) {
        // Sekme yönetimi
        std::string tab1 = browser.CreateTab("https://google.com");
        std::string tab2 = browser.CreateTab("https://github.com");
        
        // Yer işareti ekleme
        browser.AddBookmark("Google", "https://google.com");
        browser.AddBookmark("GitHub", "https://github.com");
        
        // Geçmiş araması
        auto history = browser.SearchHistory("github.com");
        
        // Navigasyon
        browser.NavigateTab(tab1, "https://example.com");
        browser.GoBackInTab(tab1);
        browser.GoForwardInTab(tab1);
        
        // Event handlers'ı ayarla
        browser.SetTabAddedCallback([](const std::string& tab_id) {
            std::cout << "Tab added: " << tab_id << std::endl;
        });
        
        // Browser'ı çalıştır
        browser.Run();
    }
    
    return 0;
}

Örnekler

Browser API kullanımı için pratik örnekler.

Modern Tarayıcı Örneği

#include "Includes/Core/comper_browser.h"

using namespace ComperCore;

class ModernBrowser : public Browser {
private:
    std::string active_tab_id;
    
public:
    bool Initialize() {
        if (!Browser::Initialize()) {
            return false;
        }
        
        // Event handlers'ı ayarla
        SetupEventHandlers();
        
        // Varsayılan ayarlar
        SetDefaultHomePage("https://www.google.com");
        SetSearchEngine("Google", "https://www.google.com/search?q=");
        
        return true;
    }
    
    void CreateInitialTabs() {
        // Ana sayfa sekmesi
        active_tab_id = CreateTab("https://www.google.com");
        
        // Popüler siteler
        CreateTab("https://github.com");
        CreateTab("https://stackoverflow.com");
        
        // Aktif et
        ActivateTab(active_tab_id);
    }
    
private:
    void SetupEventHandlers() {
        // Tab olayları
        SetTabAddedCallback([this](const std::string& tab_id) {
            std::cout << "📑 Yeni sekme: " << tab_id << std::endl;
            UpdateTabBar();
        });
        
        SetTabRemovedCallback([this](const std::string& tab_id) {
            std::cout << "❌ Sekme kapatıldı: " << tab_id << std::endl;
            UpdateTabBar();
        });
        
        SetTabActivatedCallback([this](const std::string& tab_id) {
            std::cout << "🎯 Sekme aktif: " << tab_id << std::endl;
            active_tab_id = tab_id;
            UpdateTabBar();
        });
        
        // URL değişimi olayı
        SetURLChangedCallback([this](const std::string& url) {
            std::cout << "🌐 URL değişti: " << url << std::endl;
            UpdateAddressBar(url);
        });
        
        // Yer işareti olayları
        SetBookmarkAddedCallback([this](const std::string& bookmark_id) {
            std::cout << "⭐ Yer işareti eklendi: " << bookmark_id << std::endl;
            UpdateBookmarkBar();
        });
        
        // Geçmiş olayları
        SetHistoryAddedCallback([this](const std::string& history_id) {
            std::cout << "📚 Geçmiş eklendi: " << history_id << std::endl;
            UpdateHistoryPanel();
        });
    }
    
    void UpdateTabBar() {
        // Tab bar'ı güncelle
        auto tabs = GetTabsInOrder();
        std::cout << "📑 Sekmeler (" << tabs.size() << "):" << std::endl;
        
        for (size_t i = 0; i < tabs.size(); ++i) {
            auto tab = tabs[i];
            std::string prefix = (tab->id == active_tab_id) ? "► " : "  ";
            std::cout << prefix << (i + 1) << ". " << tab->GetFormattedTitle() << std::endl;
        }
    }
    
    void UpdateAddressBar(const std::string& url) {
        // Adres çubuğunu güncelle
        std::cout << "🌐 Adres: " << url << std::endl;
    }
    
    void UpdateBookmarkBar() {
        // Yer işareti çubuğunu güncelle
        auto bookmarks = GetBookmarksInFolder("root");
        std::cout << "⭐ Yer İşaretleri (" << bookmarks.size() << "):" << std::endl;
        
        for (const auto& bookmark : bookmarks) {
            std::cout << "  - " << bookmark->title << " (" << bookmark->url << ")" << std::endl;
        }
    }
    
    void UpdateHistoryPanel() {
        // Geçmiş panelini güncelle
        auto history = GetHistory(10);
        std::cout << "📚 Son Geçmiş (" << history.size() << "):" << std::endl;
        
        for (const auto& entry : history) {
            std::cout << "  - " << entry->title << " (" << entry->url << ")" << std::endl;
        }
    }
};

int main() {
    ModernBrowser browser;
    
    if (browser.Initialize()) {
        // Başlangıç sekmelerini oluştur
        browser.CreateInitialTabs();
        
        // Browser'ı çalıştır
        browser.Run();
    }
    
    return 0;
}

Tab Yönetimi Örneği

#include "Includes/Core/comper_browser.h"

using namespace ComperCore;

class TabManagerDemo {
private:
    Browser browser;
    
public:
    bool Initialize() {
        if (!browser.Initialize()) {
            return false;
        }
        
        // Event handlers'ı ayarla
        browser.SetTabAddedCallback([](const std::string& tab_id) {
            std::cout << "📑 Sekme eklendi: " << tab_id << std::endl;
        });
        
        browser.SetTabActivatedCallback([](const std::string& tab_id) {
            std::cout << "🎯 Sekme aktif: " << tab_id << std::endl;
        });
        
        return true;
    }
    
    void RunDemo() {
        std::cout << "=== Tab Yönetimi Demo ===" << std::endl;
        
        // Sekmeleri oluştur
        std::string tab1 = browser.CreateTab("https://google.com");
        std::string tab2 = browser.CreateTab("https://github.com");
        std::string tab3 = browser.CreateTab("https://stackoverflow.com");
        
        std::cout << "\n📑 Oluşturulan sekmeler:" << std::endl;
        std::cout << "1. " << tab1 << " (Google)" << std::endl;
        std::cout << "2. " << tab2 << " (GitHub)" << std::endl;
        std::cout << "3. " << tab3 << " (Stack Overflow)" << std::endl;
        
        // Sekme navigasyonu
        std::cout << "\n🔄 Navigasyon testi:" << std::endl;
        browser.NavigateTab(tab1, "https://example.com");
        std::cout << "Tab1 navigated to example.com" << std::endl;
        
        browser.GoBackInTab(tab1);
        std::cout << "Tab1 went back to Google" << std::endl;
        
        browser.GoForwardInTab(tab1);
        std::cout << "Tab1 went forward to example.com" << std::endl;
        
        // Sekme kopyalama
        std::string tab4 = browser.DuplicateTab(tab1);
        std::cout << "\n📋 Sekme kopyalandı: " << tab4 << std::endl;
        
        // Sekme sabitleme
        browser.PinTab(tab1, true);
        std::cout << "📌 Tab1 sabitlendi" << std::endl;
        
        // Sekme sessize alma
        browser.MuteTab(tab2, true);
        std::cout << "🔇 Tab2 sessize alındı" << std::endl;
        
        // Sekme taşıma
        browser.MoveTab(tab3, 1);
        std::cout << "↔️ Tab3 pozisyon 1'e taşındı" << std::endl;
        
        // Sekme aktivasyonu
        std::cout << "\n🎯 Sekme aktivasyonu:" << std::endl;
        browser.ActivateTab(tab1);
        std::this_thread::sleep_for(std::chrono::milliseconds(500));
        
        browser.ActivateTab(tab2);
        std::this_thread::sleep_for(std::chrono::milliseconds(500));
        
        browser.ActivateTab(tab3);
        std::this_thread::sleep_for(std::chrono::milliseconds(500));
        
        // Sekmeleri kapatma
        std::cout << "\n❌ Sekmeleri kapatma:" << std::endl;
        browser.CloseTab(tab4);
        std::cout << "Kopyalanan sekme kapatıldı" << std::endl;
        
        browser.CloseTab(tab3);
        std::cout << "Stack Overflow sekmesi kapatıldı" << std::endl;
        
        browser.CloseTab(tab2);
        std::cout << "GitHub sekmesi kapatıldı" << std::endl;
        
        browser.CloseTab(tab1);
        std::cout << "Google sekmesi kapatıldı" << std::endl;
        
        std::cout << "\n✅ Tab yönetimi demo tamamlandı!" << std::endl;
    }
};

int main() {
    TabManagerDemo demo;
    
    if (demo.Initialize()) {
        demo.RunDemo();
    }
    
    return 0;
}

Yer İşareti ve Geçmiş Örneği

#include "Includes/Core/comper_browser.h"

using namespace ComperCore;

class BookmarkHistoryDemo {
private:
    Browser browser;
    
public:
    bool Initialize() {
        if (!browser.Initialize()) {
            return false;
        }
        
        // Event handlers'ı ayarla
        browser.SetBookmarkAddedCallback([](const std::string& bookmark_id) {
            std::cout << "⭐ Yer işareti eklendi: " << bookmark_id << std::endl;
        });
        
        browser.SetHistoryAddedCallback([](const std::string& history_id) {
            std::cout << "📚 Geçmiş eklendi: " << history_id << std::endl;
        });
        
        return true;
    }
    
    void RunDemo() {
        std::cout << "=== Yer İşareti ve Geçmiş Demo ===" << std::endl;
        
        // Klasörler oluştur
        std::string dev_folder = browser.CreateBookmarkFolder("Geliştirme");
        std::string social_folder = browser.CreateBookmarkFolder("Sosyal Medya");
        std::string news_folder = browser.CreateBookmarkFolder("Haberler");
        
        std::cout << "\n📁 Klasörler oluşturuldu:" << std::endl;
        std::cout << "- Geliştirme: " << dev_folder << std::endl;
        std::cout << "- Sosyal Medya: " << social_folder << std::endl;
        std::cout << "- Haberler: " << news_folder << std::endl;
        
        // Yer işaretleri ekle
        std::cout << "\n⭐ Yer işaretleri ekleniyor:" << std::endl;
        
        // Geliştirme klasörü
        std::string bm1 = browser.AddBookmark("GitHub", "https://github.com", dev_folder);
        std::string bm2 = browser.AddBookmark("Stack Overflow", "https://stackoverflow.com", dev_folder);
        std::string bm3 = browser.AddBookmark("MDN Web Docs", "https://developer.mozilla.org", dev_folder);
        
        // Sosyal medya klasörü
        std::string bm4 = browser.AddBookmark("Twitter", "https://twitter.com", social_folder);
        std::string bm5 = browser.AddBookmark("LinkedIn", "https://linkedin.com", social_folder);
        
        // Haberler klasörü
        std::string bm6 = browser.AddBookmark("BBC News", "https://bbc.com/news", news_folder);
        std::string bm7 = browser.AddBookmark("CNN", "https://cnn.com", news_folder);
        
        // Kök klasöre ekle
        std::string bm8 = browser.AddBookmark("Google", "https://google.com");
        std::string bm9 = browser.AddBookmark("YouTube", "https://youtube.com");
        
        std::cout << "Toplam 9 yer işareti eklendi" << std::endl;
        
        // Yer işareti araması
        std::cout << "\n🔍 Yer işareti araması:" << std::endl;
        auto dev_results = browser.SearchBookmarks("GitHub");
        std::cout << "GitHub araması sonuçları (" << dev_results.size() << "):" << std::endl;
        for (const auto& result : dev_results) {
            std::cout << "  - " << result->title << " (" << result->url << ")" << std::endl;
        }
        
        auto social_results = browser.SearchBookmarks("social");
        std::cout << "\nSocial araması sonuçları (" << social_results.size() << "):" << std::endl;
        for (const auto& result : social_results) {
            std::cout << "  - " << result->title << " (" << result->url << ")" << std::endl;
        }
        
        // Geçmiş oluşturma
        std::cout << "\n📚 Geçmiş oluşturma:" << std::endl;
        std::string tab = browser.CreateTab("https://example.com");
        
        // Sayfaları gezerek geçmiş oluştur
        std::vector<std::string> urls = {
            "https://google.com",
            "https://github.com",
            "https://stackoverflow.com",
            "https://developer.mozilla.org",
            "https://twitter.com",
            "https://linkedin.com",
            "https://bbc.com/news",
            "https://cnn.com",
            "https://youtube.com"
        };
        
        for (const auto& url : urls) {
            browser.NavigateTab(tab, url);
            std::this_thread::sleep_for(std::chrono::milliseconds(100));
        }
        
        // Geçmiş araması
        std::cout << "\n🔍 Geçmiş araması:" << std::endl;
        auto history_results = browser.SearchHistory("github");
        std::cout << "GitHub geçmiş araması sonuçları (" << history_results.size() << "):" << std::endl;
        for (const auto& result : history_results) {
            std::cout << "  - " << result->title << " (" << result->url << ")" << std::endl;
        }
        
        // Yer işareti güncelleme
        std::cout << "\n✏️ Yer işareti güncelleme:" << std::endl;
        browser.UpdateBookmark(bm1, "GitHub - Code Repository", "https://github.com");
        std::cout << "GitHub yer işareti güncellendi" << std::endl;
        
        // Yer işareti silme
        browser.RemoveBookmark(bm5);
        std::cout << "LinkedIn yer işareti silindi" << std::endl;
        
        // İstatistikler
        std::cout << "\n📊 Son İstatistikler:" << std::endl;
        auto all_bookmarks = browser.GetAllBookmarks();
        auto all_history = browser.GetHistory();
        
        std::cout << "Toplam yer işareti: " << all_bookmarks.size() << std::endl;
        std::cout << "Toplam geçmiş girdisi: " << all_history.size() << std::endl;
        
        std::cout << "\n✅ Yer işareti ve geçmiş demo tamamlandı!" << std::endl;
    }
};

int main() {
    BookmarkHistoryDemo demo;
    
    if (demo.Initialize()) {
        demo.RunDemo();
    }
    
    return 0;
}
