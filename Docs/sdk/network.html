Network API Genel Bakış

Comper Engine Network API, HTTP/HTTPS istekleri, WebSocket bağlantıları ve ağ yönetimi için güçlü bir arayüz sağlar.

Temel Kullanım

#include "Includes/Core/comper_network.h"

using namespace ComperCore;

int main() {
    NetworkManager network;
    
    // Network manager'ı başlat
    if (network.Initialize()) {
        // Senkron GET isteği
        NetworkResponse response = network.Get("https://api.example.com/data");
        
        if (response.IsSuccess()) {
            std::cout << "Response: " << response.body << std::endl;
            std::cout << "Status: " << response.status_code << std::endl;
        }
        
        // Asenkron GET isteği
        network.AsyncGet("https://api.example.com/async", 
            [](const NetworkResponse& response) {
                if (response.IsSuccess()) {
                    std::cout << "Async response: " << response.body << std::endl;
                }
            });
        
        // POST isteği
        std::string post_data = R"({"key": "value"})";
        NetworkResponse post_response = network.Post("https://api.example.com/submit", post_data);
        
        // Event loop (basit)
        std::this_thread::sleep_for(std::chrono::seconds(5));
    }
    
    return 0;
}

Örnekler

Network API kullanımı için pratik örnekler.

Basik API Kullanımı

#include "Includes/Core/comper_network.h"

using namespace ComperCore;

int main() {
    NetworkManager network;
    
    // Network manager'ı başlat
    if (network.Initialize()) {
        // Varsayılan ayarlar
        network.SetTimeout(30); // 30 saniye timeout
        network.SetUserAgent("ComperEngine/1.0");
        
        // Header'ları ayarla
        std::map<std::string, std::string> headers = {
            {"Accept", "application/json"},
            {"Authorization", "Bearer token123"}
        };
        network.SetDefaultHeaders(headers);
        
        // GET isteği
        NetworkResponse response = network.Get("https://api.example.com/users");
        
        if (response.IsSuccess()) {
            std::cout << "Users: " << response.body << std::endl;
        } else {
            std::cout << "Error: " << response.error_message << std::endl;
        }
        
        // POST isteği
        std::string user_data = R"({"name": "John", "email": "john@example.com"})";
        NetworkResponse post_response = network.Post("https://api.example.com/users", user_data);
        
        if (post_response.IsSuccess()) {
            std::cout << "User created: " << post_response.body << std::endl;
        }
    }
    
    return 0;
}

Asenkron İstekler

#include "Includes/Core/comper_network.h"

using namespace ComperCore;

int main() {
    NetworkManager network;
    
    if (network.Initialize()) {
        // Asenkron GET isteği
        network.AsyncGet("https://api.example.com/data", 
            [](const NetworkResponse& response) {
                if (response.IsSuccess()) {
                    std::cout << "Data received: " << response.body << std::endl;
                } else {
                    std::cout << "Error: " << response.error_message << std::endl;
                }
            });
        
        // Asenkron POST isteği
        std::string post_data = R"({"action": "update"})";
        network.AsyncPost("https://api.example.com/update", post_data,
            [](const NetworkResponse& response) {
                if (response.IsSuccess()) {
                    std::cout << "Update successful: " << response.body << std::endl;
                } else {
                    std::cout << "Update failed: " << response.error_message << std::endl;
                }
            });
        
        // Event loop (basit)
        std::this_thread::sleep_for(std::chrono::seconds(10));
    }
    
    return 0;
}

JSON API Entegrasyonu

#include "Includes/Core/comper_network.h"
#include 

using namespace ComperCore;
using json = nlohmann::json;

int main() {
    NetworkManager network;
    
    if (network.Initialize()) {
        // JSON header'ı
        network.SetDefaultHeaders({
            {"Content-Type", "application/json"},
            {"Accept", "application/json"}
        });
        
        // JSON verisi gönder
        json request_data = {
            {"query", "Comper Engine"},
            {"limit", 10},
            {"filters", {"browser", "sdk"}}
        };
        
        std::string json_string = request_data.dump();
        NetworkResponse response = network.Post("https://api.example.com/search", json_string);
        
        if (response.IsSuccess()) {
            try {
                // JSON yanıtı parse et
                json response_json = json::parse(response.body);
                
                if (response_json.contains("results")) {
                    auto results = response_json["results"];
                    std::cout << "Found " << results.size() << " results:" << std::endl;
                    
                    for (const auto& result : results) {
                        std::cout << "- " << result["title"] << std::endl;
                        std::cout << "  " << result["url"] << std::endl;
                    }
                }
            } catch (const json::exception& e) {
                std::cout << "JSON parse error: " << e.what() << std::endl;
            }
        }
    }
    
    return 0;
}

File Download

#include "Includes/Core/comper_network.h"
#include 

using namespace ComperCore;

int main() {
    NetworkManager network;
    
    if (network.Initialize()) {
        // Dosya indir
        std::string file_url = "https://example.com/file.pdf";
        NetworkResponse response = network.Get(file_url);
        
        if (response.IsSuccess()) {
            // Dosyaya yaz
            std::ofstream file("downloaded_file.pdf", std::ios::binary);
            if (file.is_open()) {
                file.write(response.body.c_str(), response.body.size());
                file.close();
                std::cout << "File downloaded successfully!" << std::endl;
                std::cout << "Size: " << response.content_length << " bytes" << std::endl;
            } else {
                std::cout << "Failed to create file!" << std::endl;
            }
        } else {
            std::cout << "Download failed: " << response.error_message << std::endl;
        }
    }
    
    return 0;
}
