Download API Genel Bakış

Comper Engine Download API, multi-threaded indirme yönetimi, pause/resume desteği ve güvenlik kontrolleri için güçlü bir arayüz sağlar.

Temel Kullanım

#include "Includes/Core/comper_download.h"

using namespace ComperCore;

int main() {
    DownloadManager download_manager;
    
    // Download manager'ı başlat
    if (download_manager.Initialize()) {
        // İndirme ekleme
        std::string download_id = download_manager.AddDownload(
            "https://example.com/largefile.zip", 
            "download.zip", 
            DownloadPriority::HIGH
        );
        
        // Progress takibi
        download_manager.on_download_progress = 
            [](const DownloadItem& download) {
                std::cout << "Progress: " 
                          << download.GetFormattedProgress() 
                          << " (" << download.GetSpeedFormatted() << ")" << std::endl;
            };
        
        // Tamamlanma olayı
        download_manager.on_download_completed = 
            [](const DownloadItem& download) {
                std::cout << "Download completed: " << download.filename << std::endl;
            };
        
        // İndirme kontrolü
        auto download = download_manager.GetDownload(download_id);
        if (download && download->IsDownloading()) {
            // Pause/Resume test
            download->Pause();
            std::this_thread::sleep_for(std::chrono::seconds(2));
            download->Resume();
        }
        
        // Event loop
        download_manager.RunEventLoop();
    }
    
    return 0;
}

Multi-Threaded İndirme Örneği

#include "Includes/Core/comper_download.h"

using namespace ComperCore;

class MultiThreadedDownloadDemo {
private:
    DownloadManager download_manager;
    
public:
    bool Initialize() {
        if (!download_manager.Initialize()) {
            return false;
        }
        
        // Konfigürasyon
        download_manager.SetMaxConcurrentDownloads(4);
        download_manager.SetDownloadDirectory("./downloads");
        
        // Event handlers'ı ayarla
        SetupEventHandlers();
        
        return true;
    }
    
    void RunDemo() {
        std::cout << "=== Multi-Threaded İndirme Demo ===" << std::endl;
        
        // Çoklu indirme ekle
        std::vector<std::string> urls = {
            "https://example.com/file1.zip",
            "https://example.com/file2.zip",
            "https://example.com/file3.zip",
            "https://example.com/file4.zip",
            "https://example.com/file5.zip"
        };
        
        std::vector<std::string> download_ids;
        
        std::cout << "\n📥 İndirmeler ekleniyor..." << std::endl;
        for (size_t i = 0; i < urls.size(); ++i) {
            std::string filename = "file" + std::to_string(i + 1) + ".zip";
            DownloadPriority priority = (i < 2) ? DownloadPriority::HIGH : DownloadPriority::NORMAL;
            
            std::string download_id = download_manager.AddDownload(urls[i], filename, priority);
            download_ids.push_back(download_id);
            
            std::cout << "İndirme eklendi: " << filename << " (ID: " << download_id << ")" << std::endl;
        }
        
        // İndirme takibi
        std::cout << "\n⏱️ İndirme takibi başlatıldı..." << std::endl;
        
        // Belirli süre sonra bazı işlemler yap
        std::this_thread::sleep_for(std::chrono::seconds(5));
        
        // İlk indirmeyi duraklat
        if (!download_ids.empty()) {
            auto download = download_manager.GetDownload(download_ids[0]);
            if (download && download->IsDownloading()) {
                download->Pause();
                std::cout << "\n⏸️ İlk indirme duraklatıldı: " << download->filename << std::endl;
                
                // 2 saniye sonra devam ettir
                std::this_thread::sleep_for(std::chrono::seconds(2));
                download->Resume();
                std::cout << "▶️ İndirme devam ettirildi: " << download->filename << std::endl;
            }
        }
        
        // İstatistikleri göster
        std::this_thread::sleep_for(std::chrono::seconds(3));
        ShowStatistics();
        
        // Tüm indirmelerin bitmesini bekle
        std::cout << "\n⏳ İndirmelerin bitmesi bekleniyor..." << std::endl;
        while (true) {
            auto downloads = download_manager.GetAllDownloads();
            bool all_completed = true;
            
            for (const auto& download : downloads) {
                if (download->IsDownloading() || download->IsPaused()) {
                    all_completed = false;
                    break;
                }
            }
            
            if (all_completed) {
                break;
            }
            
            std::this_thread::sleep_for(std::chrono::seconds(1));
        }
        
        std::cout << "\n✅ Tüm indirmeler tamamlandı!" << std::endl;
        ShowFinalStatistics();
    }
    
private:
    void SetupEventHandlers() {
        // Progress olayı
        download_manager.on_download_progress = 
            [](const DownloadItem& download) {
                std::cout << "📊 " << download.filename 
                          << ": " << download.GetFormattedProgress()
                          << " (" << download.GetSpeedFormatted() << ")"
                          << " - " << download.GetTimeRemainingFormatted() << std::endl;
            };
        
        // Tamamlanma olayı
        download_manager.on_download_completed = 
            [](const DownloadItem& download) {
                std::cout << "✅ İndirme tamamlandı: " << download.filename 
                          << " (" << download.GetFileSizeFormatted() << ")" << std::endl;
            };
        
        // Hata olayı
        download_manager.on_download_error = 
            [](const DownloadItem& download) {
                std::cout << "❌ İndirme hatası: " << download.filename 
                          << " - " << download.error_message << std::endl;
            };
        
        // Duraklatma olayı
        download_manager.on_download_paused = 
            [](const DownloadItem& download) {
                std::cout << "⏸️ İndirme duraklatıldı: " << download.filename << std::endl;
            };
        
        // Devam ettirme olayı
        download_manager.on_download_resumed = 
            [](const DownloadItem& download) {
                std::cout << "▶️ İndirme devam ettirildi: " << download.filename << std::endl;
            };
    }
    
    void ShowStatistics() {
        auto stats = download_manager.GetStatistics();
        auto downloads = download_manager.GetAllDownloads();
        
        std::cout << "\n📊 Mevcut İstatistikler:" << std::endl;
        std::cout << "Toplam indirme: " << downloads.size() << std::endl;
        std::cout << "Aktif indirme: " << stats.active_downloads << std::endl;
        std::cout << "Duraklatılmış: " << stats.paused_downloads << std::endl;
        std::cout << "Tamamlanmış: " << stats.completed_downloads << std::endl;
        std::cout << "Başarısız: " << stats.failed_downloads << std::endl;
        std::cout << "Toplam hız: " << DownloadUtils::FormatSpeed(stats.total_speed) << std::endl;
        std::cout << "Toplam boyut: " << DownloadUtils::FormatFileSize(stats.total_bytes) << std::endl;
    }
    
    void ShowFinalStatistics() {
        auto stats = download_manager.GetStatistics();
        
        std::cout << "\n📊 Final İstatistikler:" << std::endl;
        std::cout << "Toplam indirme: " << stats.total_downloads << std::endl;
        std::cout << "Tamamlanmış: " << stats.completed_downloads << std::endl;
        std::cout << "Başarısız: " << stats.failed_downloads << std::endl;
        std::cout << "Toplam boyut: " << DownloadUtils::FormatFileSize(stats.total_bytes) << std::endl;
        std::cout << "Toplam süre: " << DownloadUtils::FormatDuration(stats.total_time) << std::endl;
        std::cout << "Ortalama hız: " << DownloadUtils::FormatSpeed(stats.average_speed) << std::endl;
    }
};

int main() {
    MultiThreadedDownloadDemo demo;
    
    if (demo.Initialize()) {
        demo.RunDemo();
    }
    
    return 0;
}

İndirme Sayfası UI Örneği

#include "Includes/Core/comper_download.h"

using namespace ComperCore;

class DownloadPageDemo {
private:
    DownloadManager download_manager;
    DownloadPage download_page;
    
public:
    bool Initialize() {
        if (!download_manager.Initialize()) {
            return false;
        }
        
        // Download page'i oluştur
        download_page = DownloadPage(download_manager);
        
        // Event handlers'ı ayarla
        SetupEventHandlers();
        
        return true;
    }
    
    void RunDemo() {
        std::cout << "=== İndirme Sayfası UI Demo ===" << std::endl;
        
        // İndirme sayfasını göster
        download_page.Show();
        
        // Bazı indirmeler ekle
        std::cout << "\n📥 Örnek indirmeler ekleniyor..." << std::endl;
        
        download_page.AddDownloadFromURL("https://example.com/largefile1.zip");
        download_page.AddDownloadFromURL("https://example.com/largefile2.zip");
        download_page.AddDownloadFromURL("https://example.com/largefile3.zip");
        
        // Kullanıcı etkileşimi simülasyonu
        SimulateUserInteraction();
        
        // Event loop
        std::cout << "\n🔄 Event loop başlatılıyor..." << std::endl;
        std::cout << "Çıkmak için 'q' yazın" << std::endl;
        
        std::string input;
        while (true) {
            std::cout << "\nDownloadPage> ";
            std::getline(std::cin, input);
            
            if (input == "q" || input == "quit") {
                break;
            } else if (input == "add") {
                download_page.ShowAddDownloadDialog();
            } else if (input == "settings") {
                download_page.ShowSettingsDialog();
            } else if (input == "refresh") {
                download_page.Refresh();
            } else if (input == "stats") {
                ShowStatistics();
            } else if (input == "help") {
                ShowHelp();
            } else {
                std::cout << "Bilinmeyen komut: " << input << std::endl;
                ShowHelp();
            }
        }
        
        download_page.Hide();
        std::cout << "\n👋 İndirme sayfası kapatıldı" << std::endl;
    }
    
private:
    void SetupEventHandlers() {
        // Progress olayı
        download_manager.on_download_progress = 
            [this](const DownloadItem& download) {
                // UI'ı güncelle
                download_page.Refresh();
                
                std::cout << "📊 " << download.filename 
                          << ": " << download.GetFormattedProgress() << std::endl;
            };
        
        // Tamamlanma olayı
        download_manager.on_download_completed = 
            [this](const DownloadItem& download) {
                // UI'ı güncelle
                download_page.Refresh();
                
                std::cout << "✅ İndirme tamamlandı: " << download.filename << std::endl;
                
                // Bildirim göster (simülasyon)
                ShowNotification("İndirme Tamamlandı", 
                               download.filename + " başarıyla indirildi.");
            };
        
        // Hata olayı
        download_manager.on_download_error = 
            [this](const DownloadItem& download) {
                // UI'ı güncelle
                download_page.Refresh();
                
                std::cout << "❌ İndirme hatası: " << download.filename << std::endl;
                
                // Bildirim göster (simülasyon)
                ShowNotification("İndirme Hatası", 
                               download.filename + " indirilemedi: " + download.error_message);
            };
    }
    
    void SimulateUserInteraction() {
        std::cout << "\n👤 Kullanıcı etkileşimi simülasyonu..." << std::endl;
        
        // 3 saniye sonra yeni indirme ekle
        std::thread([this]() {
            std::this_thread::sleep_for(std::chrono::seconds(3));
            download_page.AddDownloadFromURL("https://example.com/userfile.zip");
            std::cout << "📥 Kullanıcı yeni indirme ekledi" << std::endl;
        }).detach();
        
        // 5 saniye sonra ayarları göster
        std::thread([this]() {
            std::this_thread::sleep_for(std::chrono::seconds(5));
            download_page.ShowSettingsDialog();
            std::cout << "⚙️ Kullanıcı ayarları açtı" << std::endl;
        }).detach();
        
        // 7 saniye sonra sayfayı yenile
        std::thread([this]() {
            std::this_thread::sleep_for(std::chrono::seconds(7));
            download_page.Refresh();
            std::cout << "🔄 Kullanıcı sayfayı yeniledi" << std::endl;
        }).detach();
    }
    
    void ShowNotification(const std::string& title, const std::string& message) {
        std::cout << "🔔 Bildirim: " << title << " - " << message << std::endl;
    }
    
    void ShowStatistics() {
        auto stats = download_manager.GetStatistics();
        auto downloads = download_manager.GetAllDownloads();
        
        std::cout << "\n📊 İndirme İstatistikleri:" << std::endl;
        std::cout << "Toplam indirme: " << downloads.size() << std::endl;
        std::cout << "Aktif: " << stats.active_downloads << std::endl;
        std::cout << "Duraklatılmış: " << stats.paused_downloads << std::endl;
        std::cout << "Tamamlanmış: " << stats.completed_downloads << std::endl;
        std::cout << "Başarısız: " << stats.failed_downloads << std::endl;
        std::cout << "Toplam hız: " << DownloadUtils::FormatSpeed(stats.total_speed) << std::endl;
    }
    
    void ShowHelp() {
        std::cout << "\n📋 Komutlar:" << std::endl;
        std::cout << "  add      - Yeni indirme ekle" << std::endl;
        std::cout << "  settings - İndirme ayarları" << std::endl;
        std::cout << "  refresh  - Sayfayı yenile" << std::endl;
        std::cout << "  stats    - İstatistikleri göster" << std::endl;
        std::cout << "  help     - Yardım göster" << std::endl;
        std::cout << "  q/quit   - Çıkış" << std::endl;
    }
};

int main() {
    DownloadPageDemo demo;
    
    if (demo.Initialize()) {
        demo.RunDemo();
    }
    
    return 0;
}

#include "Includes/Core/comper_download.h"

using namespace ComperCore;

class SecureDownloadDemo {
private:
    DownloadManager download_manager;
    
public:
    bool Initialize() {
        if (!download_manager.Initialize()) {

    void RunDemo() {
        std::cout << "=== Güvenli İndirme Demo ===" << std::endl;

            [](const DownloadItem& download) {
                std::cout << "❌ İndirme hatası: " << download.filename 
                          << " - " << download.error_message << std::endl;
            };
        
        // Güvenlik uyarısı olayı
        download_manager.on_security_warning = 
            [](const DownloadItem& download, const std::string& warning) {
                std::cout << "⚠️ Güvenlik uyarısı: " << download.filename 
                          << " - " << warning << std::endl;
            };
    }
    
    void ShowSecurityResults() {
        auto downloads = download_manager.GetAllDownloads();
        
        std::cout << "\n🔒 Güvenlik Sonuçları:" << std::endl;
        
        int safe_count = 0;
        int unsafe_count = 0;
        int verified_count = 0;
        int unverified_count = 0;
        
        for (const auto& download : downloads) {
            std::cout << "\n📁 " << download.filename << std::endl;
            std::cout << "  Boyut: " << download->GetFileSizeFormatted() << std::endl;
            std::cout << "  Durum: " << GetStatusString(download->status) << std::endl;
            std::cout << "  Doğrulama: " << (download->is_verified ? "✅ Tamam" : "❌ Başarısız") << std::endl;
            std::cout << "  Güvenlik: " << (download->is_safe ? "🛡️ Güvenli" : "⚠️ Tehlikeli") << std::endl;
            
            if (download->is_safe) {
                safe_count++;
            } else {
                unsafe_count++;
            }
            
            if (download->is_verified) {
                verified_count++;
            } else {
                unverified_count++;
            }
        }
        
        std::cout << "\n📊 Özet:" << std::endl;
        std::cout << "Güvenli dosyalar: " << safe_count << std::endl;
        std::cout << "Tehlikeli dosyalar: " << unsafe_count << std::endl;
        std::cout << "Doğrulanmış dosyalar: " << verified_count << std::endl;
        std::cout << "Doğrulanamamış dosyalar: " << unverified_count << std::endl;
        
        if (unsafe_count > 0) {
            std::cout << "\n⚠️ UYARI: " << unsafe_count << " adet tehlikeli dosya tespit edildi!" << std::endl;
            std::cout << "Bu dosyalar sistem için risk oluşturabilir." << std::endl;
        }
    }
    
    std::string GetStatusString(DownloadStatus status) {
        switch (status) {
            case DownloadStatus::PENDING: return "⏳ Beklemede";
            case DownloadStatus::DOWNLOADING: return "📥 İndiriliyor";
            case DownloadStatus::PAUSED: return "⏸️ Duraklatıldı";
            case DownloadStatus::COMPLETED: return "✅ Tamamlandı";
            case DownloadStatus::FAILED: return "❌ Başarısız";
            case DownloadStatus::CANCELLED: return "🚫 İptal edildi";
            case DownloadStatus::VERIFYING: return "🔍 Doğrulanıyor";
            case DownloadStatus::SCANNING: return "🛡️ Taraniyor";
            default: return "❓ Bilinmeyen";
        }
    }
};
