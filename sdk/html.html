HTML API Genel Bakış

Comper Engine HTML API, HTML5 parsing, DOM manipulation, CSS processing ve layout engine için güçlü bir arayüz sağlar.

Temel Kullanım

#include "Includes/Core/comper_html.h"

using namespace ComperCore;

int main() {
    HTMLParser parser;
    
    // HTML parse et
    std::string html = R"(
        <!DOCTYPE html>
        <html>
        <head>
            <title>Comper Engine Demo</title>
            <style>
                body { font-family: Arial, sans-serif; }
                .container { max-width: 800px; margin: 0 auto; }
                .header { background: #E60000; color: white; padding: 20px; }
            </style>
        </head>
        <body>
            <div class="container">
                <div class="header">
                    <h1>Hello Comper Engine!</h1>
                    <p>Modern HTML parsing demo</p>
                </div>
            </div>
        </body>
        </html>
    )";
    
    auto document = parser.Parse(html);
    
    if (document) {
        std::cout << "HTML parsed successfully!" << std::endl;
        
        // Element seçimi
        auto header = document->QuerySelector(".header h1");
        if (header) {
            std::cout << "Header text: " << header->GetTextContent() << std::endl;
        }
        
        // Yeni element oluşturma
        auto new_div = document->CreateElement("div");
        new_div->SetAttribute("class", "dynamic-content");
        new_div->SetTextContent("Dinamik içerik");
        
        // Document'e ekle
        auto container = document->QuerySelector(".container");
        if (container) {
            container->AppendChild(new_div);
        }
        
        // CSS parse et
        CSSParser css_parser;
        std::string css = R"(
            .dynamic-content {
                background: linear-gradient(45deg, #FF0033, #E60000);
                color: white;
                padding: 15px;
                border-radius: 8px;
                margin: 10px 0;
                animation: pulse 2s infinite;
            }
            
            @keyframes pulse {
                0% { opacity: 1; }
                50% { opacity: 0.7; }
                100% { opacity: 1; }
            }
        )";
        
        auto stylesheet = css_parser.Parse(css);
        
        // Layout engine ile render et
        LayoutEngine layout_engine;
        layout_engine.SetDocument(document);
        layout_engine.SetStylesheet(stylesheet);
        layout_engine.CalculateLayout();
        
        std::cout << "Layout calculated!" << std::endl;
    }
    
    return 0;
}

Örnekler

HTML API kullanımı için pratik örnekler.

DOM Manipülasyonu

#include "Includes/Core/comper_html.h"

using namespace ComperCore;

int main() {
    HTMLParser parser;
    
    // HTML parse et
    std::string html = R"(
        <div id="container">
            <h1>Original Title</h1>
            <p>Original content</p>
        </div>
    )";
    
    auto document = parser.Parse(html);
    
    if (document) {
        // Element seçimi
        auto container = document->QuerySelector("#container");
        if (container) {
            // Başlığı değiştir
            auto h1 = container->QuerySelector("h1");
            if (h1) {
                h1->SetTextContent("Updated Title!");
                h1->SetAttribute("style", "color: #E60000;");
            }
            
            // Yeni paragraf ekle
            auto p = document->CreateElement("p");
            p->SetTextContent("Dinamik olarak eklenen paragraf!");
            p->SetAttribute("class", "dynamic");
            
            container->AppendChild(p);
            
            // Element'i kaldır
            auto old_p = container->QuerySelector("p");
            if (old_p) {
                container->RemoveChild(old_p);
            }
            
            std::cout << "DOM manipulation completed!" << std::endl;
        }
    }
    
    return 0;
}

CSS Entegrasyonu

#include "Includes/Core/comper_html.h"

using namespace ComperCore;

int main() {
    HTMLParser parser;
    CSSParser css_parser;
    
    // HTML parse et
    std::string html = R"(
        <html>
        <head>
            <title>CSS Integration Demo</title>
        </head>
        <body>
            <div class="container">
                <div class="header">
                    <h1>Styled Header</h1>
                </div>
                <div class="content">
                    <p>Styled content</p>
                </div>
            </div>
        </body>
        </html>
    )";
    
    auto document = parser.Parse(html);
    
    // CSS parse et
    std::string css = R"(
        .container {
            max-width: 800px;
            margin: 0 auto;
            padding: 20px;
            font-family: Arial, sans-serif;
        }
        
        .header {
            background: linear-gradient(45deg, #E60000, #FF0033);
            color: white;
            padding: 20px;
            border-radius: 8px;
            text-align: center;
            box-shadow: 0 4px 8px rgba(0, 0, 0, 0.3);
        }
        
        .content {
            background: #1A1A1A;
            color: #CCCCCC;
            padding: 20px;
            margin-top: 20px;
            border-radius: 8px;
            border: 1px solid #E60000;
        }
        
        h1 {
            margin: 0;
            font-size: 24px;
            text-shadow: 0 0 10px rgba(230, 0, 0, 0.8);
        }
        
        p {
            line-height: 1.6;
            margin: 10px 0;
        }
    )";
    
    auto stylesheet = css_parser.Parse(css);
    
    if (document && stylesheet) {
        // Layout engine'i kur
        LayoutEngine layout_engine;
        layout_engine.SetDocument(document);
        layout_engine.SetStylesheet(stylesheet);
        layout_engine.SetViewportSize(1024, 768);
        
        // Layout'u hesapla
        layout_engine.CalculateLayout();
        
        std::cout << "CSS integration completed!" << std::endl;
        
        // Element stillerini kontrol et
        auto header = document->QuerySelector(".header");
        if (header) {
            auto style = header->GetComputedStyle();
            std::cout << "Header background: " << style->background_color.ToString() << std::endl;
            std::cout << "Header text color: " << style->color.ToString() << std::endl;
        }
    }
    
    return 0;
}

Dinamik Web Sayfası Oluşturma

#include "Includes/Core/comper_html.h"

using namespace ComperCore;

class DynamicWebPage {
private:
    HTMLParser parser;
    CSSParser css_parser;
    LayoutEngine layout_engine;
    std::shared_ptr<DOMDocument> document;
    
public:
    bool Initialize() {
        // Boş doküman oluştur
        document = std::make_shared<DOMDocument>();
        
        // Layout engine'i kur
        layout_engine.SetDocument(document);
        layout_engine.SetViewportSize(1024, 768);
        
        return true;
    }
    
    void CreatePage() {
        // HTML yapısı oluştur
        auto html = document->CreateElement("html");
        auto head = document->CreateElement("head");
        auto body = document->CreateElement("body");
        
        // Head element'leri
        auto title = document->CreateElement("title");
        title->SetTextContent("Dynamic Web Page");
        
        auto style = document->CreateElement("style");
        std::string css = R"(
            body {
                font-family: 'Segoe UI', Tahoma, Geneva, Verdana, sans-serif;
                background: linear-gradient(135deg, #050505, #1A1A1A);
                color: #FFFFFF;
                margin: 0;
                padding: 0;
                min-height: 100vh;
            }
            
            .container {
                max-width: 1200px;
                margin: 0 auto;
                padding: 40px 20px;
            }
            
            .header {
                text-align: center;
                margin-bottom: 40px;
                padding: 30px;
                background: linear-gradient(45deg, #E60000, #FF0033);
                border-radius: 12px;
                box-shadow: 0 8px 16px rgba(230, 0, 0, 0.3);
            }
            
            .content {
                display: grid;
                grid-template-columns: repeat(auto-fit, minmax(300px, 1fr));
                gap: 20px;
                margin-bottom: 40px;
            }
            
            .card {
                background: #1A1A1A;
                border: 1px solid #E60000;
                border-radius: 8px;
                padding: 20px;
                transition: transform 0.3s ease, box-shadow 0.3s ease;
            }
            
            .card:hover {
                transform: translateY(-5px);
                box-shadow: 0 12px 24px rgba(230, 0, 0, 0.4);
                border-color: #FF0033;
            }
            
            h1 {
                font-size: 2.5rem;
                margin: 0;
                text-shadow: 0 0 20px rgba(230, 0, 0, 0.8);
            }
            
            h2 {
                font-size: 1.5rem;
                color: #E60000;
                margin: 0 0 10px 0;
            }
            
            p {
                color: #CCCCCC;
                line-height: 1.6;
                margin: 10px 0;
            }
            
            .footer {
                text-align: center;
                padding: 20px;
                border-top: 1px solid #E60000;
                margin-top: 40px;
                color: #999999;
            }
        )";
        style->SetTextContent(css);
        
        // Head'i oluştur
        head->AppendChild(title);
        head->AppendChild(style);
        html->AppendChild(head);
        
        // Body'i oluştur
        auto container = document->CreateElement("div");
        container->SetAttribute("class", "container");
        
        // Header
        auto header = document->CreateElement("div");
        header->SetAttribute("class", "header");
        
        auto h1 = document->CreateElement("h1");
        h1->SetTextContent("Comper Engine SDK");
        header->AppendChild(h1);
        
        container->AppendChild(header);
        
        // Content
        auto content = document->CreateElement("div");
        content->SetAttribute("class", "content");
        
        // Kartlar oluştur
        for (int i = 1; i <= 4; ++i) {
            auto card = document->CreateElement("div");
            card->SetAttribute("class", "card");
            
            auto h2 = document->CreateElement("h2");
            h2->SetTextContent("Özellik " + std::to_string(i));
            
            auto p = document->CreateElement("p");
            p->SetTextContent("Bu özellik Comper Engine SDK'nin sunduğu harika bir yetenektir. Modern web teknolojileri ile entegre çalışır.");
            
            card->AppendChild(h2);
            card->AppendChild(p);
            content->AppendChild(card);
        }
        
        container->AppendChild(content);
        
        // Footer
        auto footer = document->CreateElement("div");
        footer->SetAttribute("class", "footer");
        
        auto footer_p = document->CreateElement("p");
        footer_p->SetTextContent("© 2024 Comper Engine SDK - Professional Browser Development Kit");
        
        footer->AppendChild(footer_p);
        container->AppendChild(footer);
        
        body->AppendChild(container);
        html->AppendChild(body);
        
        // Document'i ayarla
        document->AppendChild(html);
        
        // CSS'i parse et ve uygula
        auto stylesheet = css_parser.Parse(css);
        layout_engine.SetStylesheet(stylesheet);
        
        // Layout'u hesapla
        layout_engine.CalculateLayout();
        
        std::cout << "Dynamic web page created!" << std::endl;
    }
    
    std::string GetHTML() {
        // Generate HTML string from DOM
        if (document) {
            return document->ToString();
        }
        return "";
    }
};

int main() {
    DynamicWebPage webpage;
    
    if (webpage.Initialize()) {
        webpage.CreatePage();
        
        // Oluşturulan HTML'i göster
        std::cout << "Generated HTML:" << std::endl;
        std::cout << webpage.GetHTML() << std::endl;
    }
    
    return 0;
}
