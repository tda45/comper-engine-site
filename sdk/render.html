Render API Genel Bakış

Comper Engine Render API, 2D/3D rendering, Night Red Ultra grafik motoru ve GPU acceleration için güçlü bir arayüz sağlar.

Temel Kullanım

#include "Includes/Core/comper_render.h"

using namespace ComperCore;

int main() {
    Renderer renderer;
    
    // Renderer'ı başlat
    if (renderer.Initialize(nullptr)) {
        // Night Red Ultra efektlerini aktif et
        NightRedUltra::EnableNeonGlow(true);
        NightRedUltra::SetNeonIntensity(0.8f);
        NightRedUltra::SetGlowColor(230, 0, 0);
        
        // Ekranı temizle
        renderer.Clear(Color(0.02f, 0.02f, 0.02f, 1.0f));
        
        // Dikdörtgen çiz
        Rectangle rect(100, 100, 200, 50);
        renderer.DrawFilledRectangle(rect, Color(0.9f, 0.0f, 0.0f, 1.0f));
        
        // Metin çiz
        renderer.DrawText("Hello Comper Engine!", Point(150, 150), 
                          Color(1.0f, 1.0f, 1.0f, 1.0f));
        
        // Ekrana gönder
        renderer.Present();
        
        // Event loop (basit)
        std::this_thread::sleep_for(std::chrono::seconds(5));
    }
    
    return 0;
}

Örnekler

Render API kullanımı için pratik örnekler.

Night Red Ultra Efektleri

#include "Includes/Core/comper_render.h"

using namespace ComperCore;

int main() {
    Renderer renderer;
    
    if (renderer.Initialize(nullptr)) {
        // Night Red Ultra efektlerini yapılandır
        NightRedUltra::EnableNeonGlow(true);
        NightRedUltra::SetNeonIntensity(0.9f);
        NightRedUltra::SetGlowColor(230, 0, 0); // Kırmızı
        NightRedUltra::Enable4KMode(true);
        NightRedUltra::SetTargetFPS(120);
        
        // Animasyonlu render loop
        float time = 0.0f;
        for (int frame = 0; frame < 120; ++frame) {
            // Ekranı temizle
            renderer.Clear(Color(0.02f, 0.02f, 0.02f, 1.0f));
            
            // Dinamik renk değişimi
            float r = 0.9f * (0.5f + 0.5f * sin(time));
            float g = 0.0f * (0.5f + 0.5f * cos(time));
            float b = 0.0f;
            
            // Neon dikdörtgen
            Rectangle rect(200, 200, 100, 100);
            renderer.DrawFilledRectangle(rect, Color(r, g, b, 1.0f));
            
            // Metin çiz
            renderer.DrawText("Neon Glow Effect!", Point(250, 250), 
                              Color(1.0f, 1.0f, 1.0f, 1.0f));
            
            // Ekrana gönder
            renderer.Present();
            
            // Zamanı güncelle
            time += 0.1f;
            
            // Frame delay
            std::this_thread::sleep_for(std::chrono::milliseconds(16)); // ~60 FPS
        }
    }
    
    return 0;
}

2D Grafik Uygulaması

#include "Includes/Core/comper_render.h"

using namespace ComperCore;

class SimpleDrawingApp {
private:
    Renderer renderer;
    bool is_running;
    
public:
    bool Initialize() {
        is_running = true;
        
        // Renderer'ı başlat
        if (!renderer.Initialize(nullptr)) {
            return false;
        }
        
        // Night Red Ultra efektlerini aktif et
        NightRedUltra::EnableNeonGlow(true);
        NightRedUltra::SetNeonIntensity(0.8f);
        NightRedUltra::SetGlowColor(230, 0, 0);
        
        return true;
    }
    
    void Run() {
        while (is_running) {
            // Ekranı temizle
            renderer.Clear(Color(0.02f, 0.02f, 0.02f, 1.0f));
            
            // Arka plan desenleri
            DrawBackground();
            
            // UI elementleri
            DrawUI();
            
            // Ekrana gönder
            renderer.Present();
            
            // Event handling (basit)
            std::this_thread::sleep_for(std::chrono::milliseconds(16));
        }
    }
    
private:
    void DrawBackground() {
        // Grid deseni
        for (int x = 0; x < 800; x += 50) {
            renderer.DrawLine(Point(x, 0), Point(x, 600), 
                           Color(0.1f, 0.1f, 0.1f, 0.5f), 1.0f);
        }
        
        for (int y = 0; y < 600; y += 50) {
            renderer.DrawLine(Point(0, y), Point(800, y), 
                           Color(0.1f, 0.1f, 0.1f, 0.5f), 1.0f);
        }
    }
    
    void DrawUI() {
        // Başlık çerçevesi
        Rectangle header_rect(50, 50, 700, 80);
        renderer.DrawRectangle(header_rect, Color(0.9f, 0.0f, 0.0f, 1.0f), 3.0f);
        
        // Başlık metni
        renderer.DrawText("Comper Engine - Drawing Demo", Point(70, 90), 
                          Color(1.0f, 1.0f, 1.0f, 1.0f));
        
        // Neon butonlar
        for (int i = 0; i < 3; ++i) {
            Rectangle button_rect(50 + i * 250, 200, 200, 60);
            
            // Buton rengi
            float r = 0.9f * (1.0f - i * 0.3f);
            float g = 0.0f;
            float b = 0.0f;
            
            renderer.DrawFilledRectangle(button_rect, Color(r, g, b, 1.0f));
            renderer.DrawRectangle(button_rect, Color(1.0f, 1.0f, 1.0f, 1.0f), 2.0f);
            
            // Buton metni
            std::string button_text = "Button " + std::to_string(i + 1);
            renderer.DrawText(button_text, Point(60 + i * 250, 230), 
                              Color(1.0f, 1.0f, 1.0f, 1.0f));
        }
        
        // Animasyonlu daire
        static float time = 0.0f;
        time += 0.05f;
        
        float x = 400 + 100 * cos(time);
        float y = 400 + 100 * sin(time);
        
        renderer.DrawCircle(Point(x, y), 30, Color(0.0f, 0.9f, 0.0f, 1.0f));
    }
};

int main() {
    SimpleDrawingApp app;
    
    if (app.Initialize()) {
        app.Run();
    }
    
    return 0;
}

HTML Rendering Entegrasyonu

#include "Includes/Core/comper_render.h"
#include "Includes/Core/comper_html.h"

using namespace ComperCore;

class HTMLRenderer {
private:
    Renderer renderer;
    
public:
    bool Initialize() {
        return renderer.Initialize(nullptr);
    }
    
    void RenderDocument(std::shared_ptr<DOMDocument> document) {
        if (!document) return;
        
        // Ekranı temizle
        renderer.Clear(Color(0.02f, 0.02f, 0.02f, 1.0f));
        
        // Night Red Ultra efektlerini aktif et
        NightRedUltra::EnableNeonGlow(true);
        NightRedUltra::SetNeonIntensity(0.7f);
        
        // DOM elementlerini render et
        RenderDOMNode(document->GetDocumentElement(), 0, 0);
        
        // Ekrana gönder
        renderer.Present();
    }
    
private:
    void RenderDOMNode(std::shared_ptr<DOMNode> node, int x, int y) {
        if (!node) return;
        
        // Element tipine göre render et
        if (node->IsElement()) {
            auto element = std::dynamic_pointer_cast<DOMElement>(node);
            
            // Element stilini al
            auto style = element->GetComputedStyle();
            
            // Background çiz
            if (style->background_color.a > 0) {
                Rectangle bg_rect(x, y, style->width, style->height);
                renderer.DrawFilledRectangle(bg_rect, style->background_color);
            }
            
            // Border çiz
            if (style->border_width > 0) {
                Rectangle border_rect(x, y, style->width, style->height);
                renderer.DrawRectangle(border_rect, style->border_color, style->border_width);
            }
            
            // Text content render et
            if (!element->GetTextContent().empty()) {
                renderer.DrawText(element->GetTextContent(), 
                              Point(x + style->padding_left, y + style->padding_top), 
                              style->color);
            }
            
            // Child elementleri render et
            int child_x = x + style->padding_left + style->border_width;
            int child_y = y + style->padding_top + style->border_width;
            
            for (const auto& child : element->GetChildren()) {
                RenderDOMNode(child, child_x, child_y);
                child_y += child->GetComputedHeight();
            }
        } else if (node->IsText()) {
            auto text_node = std::dynamic_pointer_cast<DOMText>(node);
            renderer.DrawText(text_node->GetTextContent(), Point(x, y), 
                          Color(1.0f, 1.0f, 1.0f, 1.0f));
        }
    }
};

int main() {
    HTMLRenderer html_renderer;
    
    if (html_renderer.Initialize()) {
        // HTML parse et
        HTMLParser parser;
        std::string html = R"(
            <html>
            <head><title>Comper Engine</title></head>
            <body>
                <div style="background: #E60000; color: white; padding: 20px;">
                    <h1>Hello Comper Engine!</h1>
                    <p>Night Red Ultra rendering demo</p>
                </div>
            </body>
            </html>
        )";
        
        auto document = parser.Parse(html);
        
        // Render et
        html_renderer.RenderDocument(document);
        
        // Event loop
        std::this_thread::sleep_for(std::chrono::seconds(5));
    }
    
    return 0;
}
